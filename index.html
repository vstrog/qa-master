import requests

BASE = "https://api.example.com"

def test_create_order_happy_path():
    r = requests.post(f"{BASE}/orders", json={"sku": "ABC", "qty": 1})
    assert r.status_code == 201
    data = r.json()
    assert data["status"] == "CREATED"
    # Follow the saga
    r2 = requests.get(f"{BASE}/orders/{data['id']}")
    assert r2.json()["status"] in {"CREATED","PROCESSING","DONE"}

Kafka Load Smoke ‚Äî Python producer

from confluent_kafka import Producer
import time

conf = {"bootstrap.servers": "localhost:9092"}
p = Producer(conf)

def delivery(err, msg):
    if err:
        print(f"DELIVERY_ERR: {err}")

start = time.time(); n = 10000
for i in range(n):
    p.produce("orders", key=str(i), value=f"{{'id':{i},'ts':{time.time()}}}".encode(), callback=delivery)
    if i % 1000 == 0: p.flush()

p.flush()
print("sent_per_sec=", n / (time.time() - start))

pytest contract check

import requests, jsonschema, yaml

SPEC = yaml.safe_load(open("openapi.yaml"))
ORDER = SPEC["components"]["schemas"]["Order"]

def test_order_contract():
    r = requests.post("https://api.example.com/orders", json={"sku":"ABC","qty":1})
    assert r.status_code == 201
    jsonschema.validate(instance=r.json(), schema=ORDER)
